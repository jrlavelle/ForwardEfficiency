    function escapeHtml(str) {
      return String(str ?? "")
        .replace(/&/g, "&amp;").replace(/</g, "&lt;")
        .replace(/>/g, "&gt;").replace(/"/g, "&quot;")
        .replace(/'/g, "&#039;");

    function todayISO() {
      return new Date().toISOString().slice(0, 10);
    }

    function compareDates(a, b) {
      return new Date(a + "T00:00:00") - new Date(b + "T00:00:00");
    }

    function daysFromToday(dateStr) {
      return Math.round(
        (new Date(dateStr + "T00:00:00") - new Date(todayISO() + "T00:00:00")) / 86400000
      );
    }

    function humanizeOffset(dateStr) {
      const diff = daysFromToday(dateStr);
      if (diff === 0) return "Today";
      if (diff === 1) return "Tomorrow";
      if (diff === -1) return "Yesterday";
      if (diff > 1) return `In ${diff} days`;
      return `${Math.abs(diff)} days ago`;
    }

    function formatDate(dateStr) {
      if (!dateStr) return "";
      return new Date(dateStr + "T00:00:00")
        .toLocaleDateString(undefined, { month: "short", day: "numeric", year: "numeric" });
    }

    function isOverdue(task) {
      return task.status !== "Completed" && compareDates(task.due_date, todayISO()) < 0;

      return task.status !== "Completed" &&

      const n = daysFromToday(task.next_step_date);

    }

    }

    function buildTlCard(task, mode) {
      let timing = `Due: ${formatDate(task.due_date)}`;
      if (mode === "pastDue") timing = `Due: ${formatDate(task.due_date)} · ${Math.abs(daysFromToday(task.due_date))}d overdue`;
      if (mode === "today")   timing = task.due_date === todayISO()
        ? `Due today: ${formatDate(task.due_date)}`
        : `Next step today: ${formatDate(task.next_step_date)}`;
      if (mode === "upcoming") {
        const d = daysFromToday(task.due_date);
        timing = (d >= 1 && d <= 7)
          ? `Due: ${formatDate(task.due_date)}`
          : `Next step: ${formatDate(task.next_step_date)}`;
      }
      return `
        <div class="tl-card">
          <div class="tl-card-head">
            <h5>${escapeHtml(task.title)}</h5>
            <button class="btn-primary btn-sm" type="button" onclick="openDetailModal('${escapeHtml(task.id)}')">Open</button>
          </div>
          <div class="tl-card-meta">
            ${statusChip(task.status)}
            <span class="chip">${escapeHtml(task.workflow_type)}</span>
            <span class="chip">${escapeHtml(timing)}</span>
            <span class="chip">Assignee: ${escapeHtml(task.assignee_name)}</span>
          </div>
        </div>`;
    }

    function renderTimelinePanels() {
      const enriched = tasks.map(enrichTask);
      const pastDue  = enriched.filter(isOverdue).sort((a, b) => compareDates(a.due_date, b.due_date));
      const todayArr = enriched.filter(isToday).sort((a, b) => compareDates(a.due_date, b.due_date));
      const upcoming = enriched.filter(t => !isOverdue(t) && !isToday(t) && isUpcoming(t))
        .sort((a, b) => compareDates(a.due_date, b.due_date));

      const setPanel = (badgeId, panelId, items, mode, emptyMsg) => {
        document.getElementById(badgeId).textContent = `${items.length} item${items.length === 1 ? "" : "s"}`;
        const el = document.getElementById(panelId);
        el.innerHTML = items.length
          ? items.map(t => buildTlCard(t, mode)).join("")
          : `<p class="muted">${emptyMsg}</p>`;
      };

      setPanel("pastDueBadge",  "pastDuePanel",  pastDue,  "pastDue",  "No past due workflow items.");
      setPanel("todayBadge",    "todayPanel",    todayArr, "today",    "No workflow items require action today.");
      setPanel("upcomingBadge", "upcomingPanel", upcoming, "upcoming", "No workflow items in the next 7 days.");
    }

    function getFilteredTasks() {
      const q      = document.getElementById("searchInput").value.trim().toLowerCase();
      const status = document.getElementById("statusFilter").value;
      const type   = document.getElementById("typeFilter").value;
      const person = document.getElementById("personFilter").value;
      const qf     = document.getElementById("quickFilterSelect").value;

      return tasks.map(enrichTask).filter(task => {
        if (q) {
          const haystack = [task.title, task.workflow_type, task.owner_name,
            task.assignee_name, task.description, task.status, task.priority].join(" ").toLowerCase();
          if (!haystack.includes(q)) return false;
        }
        if (status !== "all" && task.status !== status) return false;
        if (type   !== "all" && task.workflow_type !== type) return false;
        if (person !== "all" && task.owner_id !== person && task.assignee_id !== person) return false;

        if (qf === "overdue")   return isOverdue(task);
        if (qf === "today")     return isToday(task);
        if (qf === "upcoming")  return isUpcoming(task) && !isOverdue(task) && !isToday(task);
        if (qf === "completed") return task.status === "Completed";
        if (qf === "mine")      return task.assignee_id === currentUserId;
        if (qf === "owned")     return task.owner_id === currentUserId;
        return true;
      }).sort((a, b) => {
        if (a.status === "Completed" && b.status !== "Completed") return 1;
        if (a.status !== "Completed" && b.status === "Completed") return -1;
        return compareDates(a.due_date, b.due_date);
      });
    }

    function validatePayload(payload, bannerId) {
      const required = ["title", "workflow_type", "owner_id", "assignee_id", "due_date", "next_step_date"];

        return false;

      return true;

    function checkPasswordStrength(pw, barId, labelId) {
      const bar   = document.getElementById(barId);
      const label = document.getElementById(labelId);
      if (!bar || !label) return;
      let score = 0;
      if (pw.length >= 8)  score++;
      if (pw.length >= 12) score++;
      if (/[A-Z]/.test(pw)) score++;
      if (/[0-9]/.test(pw)) score++;
      if (/[^A-Za-z0-9]/.test(pw)) score++;
      const levels = [
        { pct: "0%",   color: "transparent", text: "" },
        { pct: "25%",  color: "#ef4444",     text: "Weak" },
        { pct: "50%",  color: "#f97316",     text: "Fair" },
        { pct: "75%",  color: "#eab308",     text: "Good" },
        { pct: "100%", color: "#22c55e",     text: "Strong" },
      ];
      const lvl = levels[Math.min(score, 4)];
      bar.style.width      = lvl.pct;
      bar.style.background = lvl.color;
      label.textContent    = lvl.text;
    }

    async function initializeApp() {
      bindEvents();
      populateWorkflowTypeSelects();
      document.getElementById("dueDate").value      = todayISO();
      document.getElementById("nextStepDate").value = todayISO();

      // ─── Detect a password-recovery arrival from the URL BEFORE any awaits ───
      // Supabase may put recovery info in either the query string (?recovery=true,
      // ?code=, ?type=recovery) or the URL fragment (#type=recovery&access_token=).
      const urlQuery = new URLSearchParams(window.location.search);
      const urlHash  = new URLSearchParams(window.location.hash.replace(/^#/, ""));
      inRecoveryMode = (
        urlQuery.get("recovery") === "true" ||
        urlQuery.get("type")     === "recovery" ||
        urlHash.get("type")      === "recovery"
      );

      // If the email link came back with an error (expired / already used),
      // show a clear message right away instead of spinning forever.
      const hashError = urlHash.get("error_description") || urlHash.get("error");
      if (hashError) {
        const friendly = decodeURIComponent(hashError).replace(/\+/g, " ");
        showBanner("authBanner", "error",
          "Your reset link is no longer valid: " + friendly +
          ". Please request a new password reset email.");
        // Clean the URL so the error is not re-shown on refresh.
        window.history.replaceState({}, document.title, window.location.pathname);
        inRecoveryMode = false;
        switchAuthTab("forgot");
        setLoading(false);
        return;
      }

      // Helper: show ONLY the change-password modal — not the rest of the app.
      function showRecoveryPrompt(session) {
        currentSession = session ?? currentSession;
        currentUserId  = currentSession?.user?.id ?? "";
        // Hide both views — modal sits over a clean dark backdrop.
        document.getElementById("authView").classList.add("hidden");
        document.getElementById("appView").classList.add("hidden");
        openChangePwModal();
        showBanner("changePwBanner", "success", "Enter and confirm your new password below.");
        setLoading(false);
      }

      // Register auth listener FIRST so PASSWORD_RECOVERY / SIGNED_IN aren't missed.
      db.auth.onAuthStateChange(async (event, session) => {
        // Any recovery-related event routes to the password reset prompt.
        if (event === "PASSWORD_RECOVERY" || (inRecoveryMode && session)) {
          showRecoveryPrompt(session);
          return;
        }
        if (event === "SIGNED_IN" && session && session.user?.id !== currentUserId) {
          currentSession = session;
          currentUserId  = session.user?.id ?? "";
          try {
            await ensureProfileRow(session.user);
            await enterApp();
          } catch (err) {
            showBanner("authBanner", "error", err.message ?? String(err));
          }
        }
      });

      setLoading(true);
      try {
        // Give Supabase a moment to auto-process any tokens in the URL hash.
        if (window.location.hash.includes("access_token")) {
          await new Promise(r => setTimeout(r, 250));
        }

        const { data, error } = await db.auth.getSession();
        if (error) throw error;

        currentSession = data.session;
        currentUserId  = currentSession?.user?.id ?? "";

        if (currentSession && inRecoveryMode) {
          // Arrived via reset email — only show the password-reset prompt.
          showRecoveryPrompt(currentSession);
        } else if (currentSession) {
          await ensureProfileRow(currentSession.user);
          await enterApp();
        }
      } catch (err) {
        showBanner("authBanner", "error", err.message ?? String(err));
      } finally {
        // Clean tokens out of the URL once we've parsed them.
        if (window.location.hash || window.location.search.match(/(recovery|code|type|access_token)=/)) {
          window.history.replaceState({}, document.title, window.location.pathname);
        }
        setLoading(false);
      }
    }

    initializeApp();
  </script>
